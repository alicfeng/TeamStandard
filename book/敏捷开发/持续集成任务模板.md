# 持续集成任务流模板

## 前言

如下规范模板均基于 `Drone` 为基础前提



## 模板

#### PHP

```yaml
kind: pipeline
name: demo_sample
type: docker

steps:
- name: 加载缓存
  image: drillster/drone-volume-cache
  volumes:
  - name: cache
    path: /cache
  settings:
    restore: true
    mount:
      - ./vendor

- name: 安装依赖
  image: hub.c.163.com/applications/laravle:develop
  commands:
    - COMPOSER_MEMORY_LIMIT=-1 /usr/local/bin/composer install --optimize-autoloader -vvv
  when:
    branch:
      - develop

- name: 单元测试
  image: hub.c.163.com/applications/laravle:develop
  commands:
    - ./vendor/phpunit/phpunit/phpunit

- name: 构建镜像
  image: plugins/docker
  settings:
    username:
      from_secret: DOCKER_USERNAME
    password:
      from_secret: DOCKER_PASSWORD
    repo: hub.c.163.com/applications/laravle
    registry: hub.c.163.com
    insecure: true
    purge: false
    tags: 
    - ${DRONE_BUILD_NUMBER}
    - ${DRONE_TAG}
    
- name: 部署开发
  image: quay.io/honestbee/drone-kubernetes
  settings:
    kubernetes_server:
      from_secret: KUBERNETES_DEV_SERVER
    kubernetes_token:
      from_secret: KUBERNETES_DEV_TOKEN
    namespace:
      from_secret: KUBERNETES_DEVT_NAMESPACE
    deployment: api
    container: [app]
    repo: hub.c.163.com/applications/laravle
    tag: ${DRONE_BUILD_NUMBER}

- name: 代码分析
  image: aosapps/drone-sonar-plugin
  detach: true
  settings:
    sonar_host:
      from_secret: SONAR_HOST
    sonar_token:
      from_secret: SONAR_TOKEN
  when:
    branch:
      - develop

- name: 构建文档
  image: plugins/docker
  settings:
    username:
      from_secret: DOCKER_HUB_USERNAME
    password:
      from_secret: DOCKER_HUB_PASSWORD
    repo: hub.c.163.com/applications/laravle_apidoc
    registry: hub.c.163.com
    dockerfile: Dockerfile.apidoc
    tags: ${DRONE_TAG=latest}

- name: 文档部署
  image: appleboy/drone-ssh
  detach: true
  settings:
    host:
      from_secret: DOCUMENT_SERVER_HOST
    username:
      from_secret: DOCUMENT_SERVER_USERNAME
    password:
      from_secret: DOCUMENT_SERVER_PASSWORD
    port: 22
    script:
    - docker pull hub.c.163.com/applications/laravle_apidoc:latest
    - cd /home/alicfeng/tutorial/ev/dev_tool/document
    - docker-compose up -d laravel.apidoc
      
- name: 部署测试
  image: quay.io/honestbee/drone-kubernetes
  settings:
    kubernetes_server:
      from_secret: KUBERNETES_TEST_SERVER
    kubernetes_token:
      from_secret: KUBERNETES_TEST_TOKEN
    namespace:
      from_secret: KUBERNETES_TEST_NAMESPACE
    deployment: api
    container: [app]
    repo:  hub.c.163.com/applications/laravle
    tag: ${DRONE_BUILD_NUMBER}
  when:
    target:
    - test

- name: 部署生产
  image: quay.io/honestbee/drone-kubernetes
  settings:
    kubernetes_server:
      from_secret: KUBERNETES_TEST_SERVER
    kubernetes_token:
      from_secret: KUBERNETES_TEST_TOKEN
    namespace:
      from_secret: KUBERNETES_TEST_NAMESPACE
    deployment: api
    container: [app]
    repo:  hub.c.163.com/applications/laravle
    tag: ${DRONE_BUILD_NUMBER=latest}
  when:
    target:
    - production

- name: 构建缓存
  image: drillster/drone-volume-cache
  volumes:
    - name: cache
      path: /cache
  settings:
    rebuild: true
    mount:
    - ./vendor

- name: 邮件通知
  image: drillster/drone-email
  settings:
    from: 
     from_secret: EMAIL_FROM
    host: 
     from_secret: EMAIL_HOST
    username: 
     from_secret: EMAIL_USERNAME
    password:
     from_secret: EMAIL_PASSWORD
    port: 
     from_secret: EMAIL_PORT
    recipients:
    - {a@samego.com}
  when:
    status: [ failure, success ]

volumes:
  - name: cache
    host:
      path: /data/drone/composer
```


#### 前端

```YAML
kind: pipeline
name: demo_sample
type: docker

steps:
- name: 加载缓存
  image: drillster/drone-volume-cache
  volumes:
  - name: cache
    path: /cache
  settings:
    restore: true
    mount:
    - ./node_modules

- name: 安装依赖
  image: hub.c.163.com/library/node:latest
  commands:
  - yarn config set ignore-engines true
  - yarn config set registry https://registry.npm.taobao.org/
  - yarn install -ddd
  when:
    branch:
    - develop

- name: 源码构建
  image: hub.c.163.com/library/node:latest
  commands:
  - yarn build -ddd

- name: 构建缓存
  image: drillster/drone-volume-cache
  detach: true
  volumes:
  - name: cache
    path: /cache
  settings:
    rebuild: true
    mount:
    - ./node_modules
  when:
    branch:
    - develop

- name: 构建镜像
  image: plugins/docker
  settings:
    username:
      from_secret: DOCKER_USERNAME
    password:
      from_secret: DOCKER_PASSWORD
    repo: hub.c.163.com/applications/vue
    registry: hub.c.163.com
    insecure: true
    purge: false
    tags: 
    - ${DRONE_BUILD_NUMBER}
    - ${DRONE_TAG}

- name: 部署测试
  image: quay.io/honestbee/drone-kubernetes
  settings:
    kubernetes_server:
      from_secret: KUBERNETES_TEST_SERVER
    kubernetes_token:
      from_secret: KUBERNETES_TEST_TOKEN
    namespace:
      from_secret: KUBERNETES_TEST_NAMESPACE
    deployment: client
    container: [app]
    repo:  hub.c.163.com/applications/vue
    tag: ${DRONE_BUILD_NUMBER}
  when:
    target:
    - test

- name: 部署生产
  image: quay.io/honestbee/drone-kubernetes
  settings:
    kubernetes_server:
      from_secret: KUBERNETES_TEST_SERVER
    kubernetes_token:
      from_secret: KUBERNETES_TEST_TOKEN
    namespace:
      from_secret: KUBERNETES_TEST_NAMESPACE
    deployment: client
    container: [app]
    repo:  hub.c.163.com/applications/vue
    tag: ${DRONE_BUILD_NUMBER=latest}
  when:
    target:
    - production

- name: 构建缓存
  image: drillster/drone-volume-cache
  volumes:
    - name: cache
      path: /cache
  settings:
    rebuild: true
    mount:
    - ./vendor


volumes:
- name: cache
  host:
    path: /data/drone/node_modules
```
